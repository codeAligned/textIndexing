% Pseudocode.

% Parameter 3: Input description.
% Parameter 4: Output description.
\newenvironment{pseudocode}[4]
{
  \begin{algorithm}[H]
    \caption{#1}
    \label{alg:#2}
    \setlength{\tabcolsep}{0mm}
    \begin{tabularx}{\linewidth}{>{\bfseries}p{20mm}X}
      Input  & #3 \\
      Output & #4 \\
      \midrule
    \end{tabularx}
    \begin{algorithmic}[1]
}
{
    \end{algorithmic}
  \end{algorithm}
}
